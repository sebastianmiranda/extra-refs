% tecnología
\newglossaryentry{Linux} { description={es un término general usado para referirse a la familia de sistemas operativos basados en UNIX que usan el kernel Linux}}

\newglossaryentry{Virtual Private Network} { description={es una conexión establecida desde una red externa hacia una red interna. Generalmente se hace cifrada para proteger los datos que se transmiten en la conexión}}

\newglossaryentry{OpenVPN} { description={es software libre que permite establecer conexiones VPN cifrando las conexiones con ayuda de algoritmos de OpenSSL}}

\newglossaryentry{OpenSSL} { description={es software libre que permite la conexión entre dos puntos de manera cifrada y segura. Esta tecnología es muy usada en las conexiones seguras de los navegadores de internet.}}
